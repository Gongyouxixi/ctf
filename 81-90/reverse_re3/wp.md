import hashlib

def string_to_md5(input_string):
    # 创建 MD5 哈希对象
    md5_hash = hashlib.md5()
    # 将输入字符串编码为字节类型
    input_bytes = input_string.encode('utf-8')
    # 更新哈希对象的内容
    md5_hash.update(input_bytes)
    # 获取十六进制表示的 MD5 哈希值
    return md5_hash.hexdigest()

# 示例使用
input_string = "ddsssddddsssdssdddddsssddddsssaassssdddsddssddwddssssssdddssssdddss"
md5_result = string_to_md5(input_string)
print(f"字符串 '{input_string}' 的 MD5 哈希值是: {md5_result}")
